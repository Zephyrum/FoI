		months = 36
	}

		name = "foi_syrian_revolt.1.a"
		
		ai_chance = {
			factor = 100
		}
		
		every_province = {
			limit = {
				OR = {
					culture = hebrew
					culture = palestinian
					culture = aramaic
				}
			}
			add_core = SYR
		}
		
		every_owned_province = {
			limit = {
				OR = {
					culture = hebrew
					culture = palestinian
					culture = aramaic
				}
			}
			cede_province = SYR
		}
		
		hidden_effect = {
			SYR = {
				set_country_flag = syrian_revolt_ongoing
				if = {
					limit = {
						NOT = { government = monarchy }
						is_revolutionary = no
					}
					change_government_to_monarchy = yes
				}
				if = {
					limit = { is_revolutionary = no }
					add_government_reform = dynastyic_reform_foi
				}
				if = {
					limit = { owns = 382 }
					set_capital = 382
				}
				change_government = monarchy
				adopt_reform_progress = ROOT
				add_idea_group = defensive_ideas
				add_idea = improved_foraging
				add_idea_group = economic_ideas
				add_idea = smithian_economics
				add_idea_group = diplomatic_ideas
				add_idea = diplomatic_corps
				add_idea_group = offensive_ideas
				add_idea = napoleonic_warfare
				set_government_rank = 2
				change_primary_culture = hebrew
				add_accepted_culture = aramaic
				add_accepted_culture = palestinian
				change_religion = monothelite
				add_prestige = 100
				add_legitimacy = 100
				add_adm_power = 450
				add_dip_power = 450
				add_mil_power = 450
				add_treasury = 250
				create_general = { tradition = 100 }
				if = {
					limit = { any_owned_province = { has_port = yes } }
					random_owned_province = {
						limit = { has_port = yes }
						heavy_ship = SYR
						light_ship = SYR
						light_ship = SYR
						light_ship = SYR
						light_ship = SYR
						galley = SYR
						galley = SYR
						galley = SYR
						transport = SYR
						transport = SYR
						transport = SYR
						transport = SYR
						transport = SYR
						transport = SYR
						transport = SYR
						transport = SYR
					}
				}
				capital_scope = {
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					cavalry = SYR
					cavalry = SYR
					artillery = SYR
					artillery = SYR
					artillery = SYR
					artillery = SYR
					artillery = SYR
					artillery = SYR
				}
			}
		}
		SYR = {
			define_ruler = {
				name = "Salome"
				dynasty = "Heroid"
				adm = 6
				dip = 6
				mil = 6
				age = 26
				claim = 100
				female = yes
				culture = hebrew
				religion = monothelite
			}
			define_heir = {
				dynasty = "Heroid"
				age = 5
				adm = 6
				dip = 6
				mil = 6
				claim = 100
				female = yes
				no_consort_with_heir = no
			}
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
	}
	
	option = { #join revolt
		name = "foi_syrian_revolt.1.b"
		
		ai_chance = {
			factor = 0
		}
		
		every_province = {
			limit = {
				OR = {
					culture = hebrew
					culture = palestinian
					culture = aramaic
				}
			}
			add_core = SYR
		}
		
		every_owned_province = {
			limit = {
				OR = {
					culture = hebrew
					culture = palestinian

				}

				add_idea = improved_foraging
				add_idea_group = economic_ideas
				add_idea = smithian_economics
				add_idea_group = diplomatic_ideas
				add_idea = diplomatic_corps
				add_idea_group = offensive_ideas
				add_idea = napoleonic_warfare
				set_government_rank = 2
				change_primary_culture = hebrew
				add_accepted_culture = aramaic
				add_accepted_culture = palestinian
				change_religion = monothelite
				add_prestige = 75
				add_legitimacy = 75
				add_adm_power = 150
				add_dip_power = 150
				add_mil_power = 150
				add_treasury = 150
				create_general = { tradition = 50 }
				if = {
					limit = { any_owned_province = { has_port = yes } }
					random_owned_province = {
						limit = { has_port = yes }
						heavy_ship = SYR
						light_ship = SYR
						light_ship = SYR
						light_ship = SYR
						light_ship = SYR
						galley = SYR
						galley = SYR
						galley = SYR
						transport = SYR
						transport = SYR
						transport = SYR
						transport = SYR
						transport = SYR

					}
				}
				capital_scope = {
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					infantry = SYR
					cavalry = SYR
					cavalry = SYR
					artillery = SYR
					artillery = SYR
					artillery = SYR
					artillery = SYR
					artillery = SYR
					artillery = SYR
				}
			}
		}
		SYR = {
			define_ruler = {
				name = "Salome"

				adm = 6
				dip = 6
				mil = 6
				age = 26
				claim = 100
				female = yes
				culture = hebrew
				religion = monothelite
			}
			define_heir = {
				dynasty = "Heroid"
				age = 5
				adm = 6
				dip = 6
				mil = 6
				claim = 100
				female = yes
				no_consort_with_heir = no
			}
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
		switch_tag = SYR
	}

# Lebanon Joins Revolt
country_event = {
	id = foi_syrian_revolt.2
	title = "foi_syrian_revolt.2.title"
	desc = "foi_syrian_revolt.2.desc"
	picture = CONQUEST_eventPicture
	
	trigger = {
		tag = SYR
		has_country_flag = syrian_revolt_ongoing
		exists = LEB
		LEB = {
			NOT = { total_development = 100 }
			ai = yes
			NOT = { is_rival = SYR }
			NOT = {
				OR = {
					religion = catholic
					religion_group = muslim
				}
			}
			has_opinion = {
				who = ROOT
				value = 0
			}
		}
	}
	
	mean_time_to_happen = {
		months = 4
	}
	
	immediate = {
		if = {
			limit = { LEB = { is_subject = yes NOT = { is_subject_of_type = tributary_state } } }
			every_country = {
				limit = { overlord_of = LEB }
				every_province = {
					limit = { is_core = LEB }
					add_core = PREV
				}
			}
		}
	}
	
	major = yes
	
	option = { #what a glorious day
		name = "foi_syrian_revolt.2.a"
		
		ai_chance = {
			factor = 100
		}
		
		inherit = LEB
		
	}
	
}
