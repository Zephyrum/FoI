country_event = {
	id = chinesekorean.101 # Enforcing Neo-Korean
	title = chinesekorean.101_title
	desc = chinesekorean.101_desc
	picture = MERITOCRACY_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = chinesekorean.101_O1 # Neo-Korean is enforced
		every_owned_province = {
				limit = {
					culture = korean
				}
			change_culture = neokorean
			add_province_modifier = {
				name = distorted_our_traditions
				duration = 20
			}
		}
		every_country = {
			limit = {
				primary_culture = korean
			}
			add_opinion = {
				who = ROOT
				modifier = opinion_distorted_our_traditions
				years = 100
			}
		}
		hidden_effect = {
			set_country_flag = flag_forced_chinesekorean
			every_country = {
				limit = {
					ai = no
					NOT = {
						tag = ROOT
					}
				}
				country_event = {
					id = chinesekorean.102 #Neo-Korean is enfored
				}
			}
		}
	}
	option = {
		name = chinesekorean.101_O2 # Quit enforcing Neo-Korean
		add_dip_power = 100
		add_mil_power = 200
		add_meritocracy = 20
		add_prestige = 20
	}
}
