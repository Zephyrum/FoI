#arabian transfer
country_event = {
	id = foi_flavor_rum.2
	title = none
	desc = none
	hidden = yes

	trigger = {
		has_country_flag = sjk_haasa_arabia_I_flag
		ALH = {
			is_subject_of = ROOT
		}
		ALH = {
			has_country_flag = sjk_haasa_arabia_II_flag
		}
		any_owned_province = {
			region = arabia_region
		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	
	immediate = {
		every_province = {
			limit = {
				AND = {
					owned_by = ROOT
					region = arabia_region
					is_core = ROOT
				}
			}
			cede_province = ALH
			remove_core = ROOT
			add_core = ALH
		}
		every_province = {
			limit = {
				AND = {
					owned_by = ROOT
					region = arabia_region
					NOT = { is_core = ROOT }
				}
			}
			cede_province = ALH
		}
	}
	
	option = { }
}
