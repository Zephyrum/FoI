#Death of Khagan
#This event fires regardless of the Khagan's shit being taken care of. There should be two descriptions: one for if he took care of his shit, and one for if he didn't. The effects in the immediate block already account for this dichotomy.
country_event = {
	id = foi_yuan.2
	title = foi_yuan.2.t
	desc = {
		trigger = {
			has_country_flag = foi_yuan_took_care_of_his_shit
		}
		desc = foi_yuan.2.desc_good		
	}
	desc = {
		trigger = {
			NOT = {
				has_country_flag = foi_yuan_took_care_of_his_shit
			}
		}
		desc = foi_yuan.2.desc_bad
	}
	picture = CIVIL_WAR_eventPicture

	major = yes
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		tag = YUA
		NOT = {
			has_ruler_modifier = khagan_reforms
		}
	}

	immediate = {
		clr_country_flag = foi_yuan_chain_started
		if = {
			limit = {
				NOT = {
					has_country_flag = foi_yuan_took_care_of_his_shit
				}
			}
			set_global_flag = foi_yuan_bob_died_uh_oh
			every_country = {
				limit = {
					tag = JIN
				}
				country_event = {
					id = foi_yuan.3
					days = 30
					random = 10
				}
			}	
		}
	}

	#RIP big Khaggy.
	option = {
		name = foi_yuan.2.a
	}
}

		}

#Jin at the Gates
country_event = {
	id = foi_yuan.7
	title = foi_yuan.7.t
	desc = foi_yuan.7.desc
	picture = CIVIL_WAR_eventPicture

	is_triggered_only = yes

	#Bend the knee.
	option = {
		name = foi_yuan.7.a
		set_global_flag = foi_yuan_bent_knee
		north_china_region = {
			limit = {
				owned_by = YUA
			}
			cede_province = JIN
			add_core = JIN
			remove_core = YUA
		}
		every_subject_country = {
			grant_independence = yes
		}
		hidden_effect = {
			every_country = {
				limit = {
					tag = JIN
				}
				country_event = {
					id = foi_yuan.8
				}
			}
		}
	}

	#Fight back!
	#This option brings a rebellion because the court nobles are not supportive of the new Khagan.
	option = {
		name = foi_yuan.7.b
		set_global_flag = foi_yuan_fighting_back
		capital_scope = {
			spawn_rebels = {
				size = 2
				type = noble_rebels
				friend = JIN
			}
		}
		hidden_effect = {
			every_country = {
				limit = {
					tag = JIN
				}
				country_event = {
					id = foi_yuan.9
				}
			}
		}
	}
}
