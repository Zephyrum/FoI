country_event = {
	id = flavor_wal.4
	title = "flavor_wal.4.title"
	desc = {
		trigger = {
			AND = {
				has_heir = yes
				has_female_heir = no
				heir_age = 15
			}
		}
		desc = "flavor_wal.4.desc.a"
	}
	desc = {
		trigger = {
			OR = {
				AND = {
					has_heir = yes
					has_female_heir = yes
				}
			}
		}
		desc = "flavor_wal.4.desc.b"
	}
	desc = {
		trigger = {
			OR = {
				AND = {
					has_heir = yes
					has_female_heir = no
					NOT = { heir_age = 15 }
				}
			}
		}
		desc = "flavor_wal.4.desc.c"
	}
	desc = {
		trigger = {
			OR = {
				AND = {
					has_heir = no
				}
			}
		}
		desc = "flavor_wal.4.desc.d"
	}
	picture = IMPALED_SOLDIERS_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = WAL
		exists = MOL
		MOL = { is_subject = no }
		MOL = { government = monarchy }
		MOL = { religion_group = christian }
		NOT = { is_year = 1500 }
		OR = {
			AND = {
				vassal_of = BUL
				any_country = {
					war_score_against = {
						who = BUL
						value = 60
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 8
	}
	
	immediate = {
		hidden_effect = {
			grant_independence = yes
			every_country = {
				limit = { war_with = ROOT }
				white_peace = ROOT
			}
		}
	}
	
	option = {
		name = "flavor_wal.4.a"
		ai_chance = { factor = 85 }
		custom_tooltip = wal_becomes_independant.tt
		if = {
			limit = { has_heir = yes }
			kill_heir = yes
		}
		MOL = { create_union = ROOT }
		add_prestige = 50
		MOL = { change_adm = 1 }
		MOL = { change_dip = 1 }
		MOL = { change_mil = 1 }
		BUL = {
			add_casus_belli = {
				target = ROOT
				type = cb_disloyal_vassal
				months = 300
			}
		}
		hidden_effect = {
			MOL = {
				country_event = { id = flavor_wal.41 }
			}
			BUL = {
				country_event = { id = flavor_wal.42 }
			}
		}
	}
	
	option = {
		name = "flavor_wal.4.b"
		ai_chance = { factor = 15 }
		custom_tooltip = wal_becomes_independant.tt
		if = {
			limit = { has_heir = yes }
			kill_heir = yes
		}
		define_ruler = {
			name = "Nicolae"
			dynasty = ROOT
			adm = 0
			dip = 0
			mil = 2
			fixed = yes
			age = 45
			claim = 30
		}
		add_prestige = -25
		BUL = {
			add_casus_belli = {
				target = ROOT
				type = cb_disloyal_vassal
				months = 300
			}
		}
		MOL = {
			add_casus_belli = {
				target = ROOT
				type = cb_restore_personal_union
				months = 300
			}
		}
		hidden_effect = {
			add_ruler_personality = babbling_buffoon_personality
			add_ruler_personality = cruel_personality
			add_ruler_personality = greedy_personality
			MOL = {
				country_event = { id = flavor_wal.43 }
			}
			BUL = {
				country_event = { id = flavor_wal.44 }
			}
		}
	}
