#Pala_Flavor_Event Rivalry
country_event = {
	id = flavorPLD.1
	title = flavorPLD.1.title
	desc = flavorPLD.1.desc
	picture = COURT_eventPicture

	is_triggered_only = yes

	trigger = {
		tag = PLD
		is_subject = no
		has_country_modifier = tibet_tribute_pld
	}

	option = { # Deny Tibetan Request
		name = "flavorPLD.1.opta"
		ai_chance = {
			factor = 50
			modifier = {
				factor = 75
				is_at_war = yes
			}
			modifier = {
				factor = 0
				has_country_flag = tibet_tribute_flag_loyal
			}
		}
		hidden_effect = {
    		country_event = {
        		id = flavorPLD.1
        		days = 360
        		random = 30
			}
			reverse_add_opinion = {
    			who = TIB
    			modifier = Deny_Tibetan_Request_Rivalry
			}
		}
		add_prestige = 16
		add_legitimacy = 18
	}
	
	option = { #Accept Tibetan Request
		name = "flavorPLD.1.optb"
		ai_chance = {
			factor = 50
			modifier = {
				factor = 25
				is_at_war = yes
			}
			modifier = {
				factor = 100
				has_country_flag = tibet_tribute_flag_loyal
			}
		}
		hidden_effect = {
			random_list = {
				60 = {
					country_event = {
        				id = flavorPLD.1
        				days = 360
        				random = 30
					}
				}
				40 = {
					country_event = {
        				id = flavorPLD.2
        				days = 360
        				random = 30
					}
				}
			}
			reverse_add_opinion = {
    			who = TIB
    			modifier = Accept_Tibetan_Request_Rivalry
			}
			TIB = {
				add_yearly_manpower = 1
			}
		}
	add_yearly_manpower = -0.2
	add_prestige = -8
	add_legitimacy = -2
	add_mil_power = 4
	}
}
