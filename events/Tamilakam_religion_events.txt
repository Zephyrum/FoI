namespace = foi

#The Kingdom of Tamilakam since its formation has been fairly zealous nationa which has caused tensions withint the kingdom amongst the Hindu population.
#The Hindu population will want to decrease the zealotry of the current goverment and will promote a more tolerant Chera noble to take control of the kingdom

country_event = {
	id = foi.5
	title = "foi.eventname5"
	desc = "foi.eventdesc5"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		if = {
			tag = TAM
			is_subject = no
			religion = nestorian
			NOT = { religious_unity = 0.5 
			}
		}
	}

	mean_time_to_happen = {
		months = 24
	}
	
	option = { #Accept
		name = "foi.event5opt1"
		ai_chance = { factor = 0.2 }

		define_ruler = {
			name = "Philipose"
			dynasty = "Chera"
			culture = malayalam
			religion = nestorian
			fixed = yes
			dip = 5
		}

		hidden_effect = { 
			add_ruler_personality = tolerant_personality

			if = {
				limit = {
					has_heir = yes
				}
				remove_heir = yes
			}

			if = {
				limit = {
					VND = {
						is_subject = yes
					}
				}
				TAM = {
					inherit = VND
				}
			}
		}

		add_country_modifier = {
			name = "tolerant_government"
			duration = -1 
		}

		add_stability = -2
	}

	
	option = { #Deny
		name = "foi.event5opt2"
		ai_chance = { factor = 0.8 }
		add_prestige = 33
		set_country_flag = tamilakam_rebellion
		random_owned_province = {
			limit = {
				development = 12
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 3
				leader = "Philipose Chera"
				culture = malayalam
				religion = nestorian
			}
		}
	}
}

country_event = { 
	id = foi.15
	title = "foi.eventname15"
	desc = "foi.eventdesc15"
	picture = ADVISOR_eventPicture

	fire_only_once = yes

	trigger = {
		if = {
			tag = TAM
			is_subject = no
			religion = nestorian
			dynasty = "Chera"
			has_country_flag = tamilakam_rebellion
		}
		NOT = {
			num_of_rebel_armies = 1
		}
	}

	mean_time_to_happen = {
		months = 1
	}

	option = { #Philipose won the civil war
		name = "foi.event15opt1"

		if = {
			limit = {
				VND = {
					is_subject = yes
				}
			}
			TAM = {
			inherit = VND
			}
		}

		hidden_effect = { 
			add_ruler_personality = tolerant_personality
			change_dip = 3
		}
		
		add_country_modifier = {
			name = "tolerant_government"
			duration = -1 
		}

		add_stability = 1
	}
}
