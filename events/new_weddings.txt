#The Polish-Lithuanian Wedding
country_event = {
	id = foi.7
	title = "foi.eventname7"
	desc = "foi.eventdesc7"
	picture = DIPLOMACY_eventPicture

	major = yes
	
	fire_only_once = yes
	
	
	trigger = {
		NOT = { exists = PLC }
		OR = {
			AND = {
				tag = POL
				NOT = { war_with = LIT }
				LIT = {
					exists = yes
					ai = yes
					was_player = no
					is_subject = no
					is_neighbor_of = ROOT
				}
			}
			AND = {
				tag = LIT
				NOT = { war_with = POL }
				ai = no
				POL = {
					exists = yes
					ai = yes
					was_player = no
					is_subject = no
					is_neighbor_of = ROOT
				}
			}
		}
		is_year = 1450
		NOT = { is_year = 1530 }
		is_subject = no
		num_of_cities = 10
		OR = {
			AND = {
				POL = { is_female = yes }
				LIT = { is_female = no }
			}
			AND = {
				POL = { is_female = no }
				LIT = { is_female = yes }
			}
			LIT = {
				ai = yes
				has_regency = yes
			}
			POL = {
				ai = yes
				has_regency = yes
			}
		}
	}

	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.1
			OR = {
				AND = {
					tag = POL
					LIT = {
						has_regency = no 
					}
				}
				AND = {
					tag = LIT
					POL = { 
						has_regency = no 
					}
				}
			}
		}
	}
	
	
	option = {		# Let us bind their Dynasty to ours
		name = "foi.event7opt1"
		ai_chance = { factor = 90 }
		if = {
			limit = { tag = POL }
			create_union = LIT
			add_opinion = {
				who = LIT
				modifier = pol_wedding
			}
			LIT = {
				add_opinion = {
					who = ROOT
					modifier = pol_wedding
				}
				add_liberty_desire = -50
			}
			every_known_country = {
				limit = {
					marriage_with = LIT
				}
				add_opinion = {
					who = ROOT
					modifier = pol_wedding_negative
				}
			}
		}
		if = {
			limit = { tag = LIT }
			create_union = POL
			add_opinion = {
				who = POL
				modifier = pol_wedding
			}
			POL = {
				add_opinion = {
					who = ROOT
					modifier = pol_wedding
				}
				add_liberty_desire = -50
			}
			every_known_country = {
				limit = {
					marriage_with = POL
				}
				add_opinion = {
					who = ROOT
					modifier = pol_wedding_negative
				}
			}
		}
		random_owned_province = {
			limit = {
				is_capital = no
			}
			spawn_rebels = {
				type = noble_rebels
				size = 3
			}
		}
	}
	option = {		# No let us marry a local talent instead
		name = "foi.event7opt2"
		ai_chance = { factor = 10 }
		add_stability_or_adm_power = yes
		if = {
			limit = { tag = POL }
			LIT = {
				add_opinion = {
					who = ROOT
					modifier = pol_declined_wedding
				}
			}
		}
		if = {
			limit = { tag = LIT }
			POL = {
				add_opinion = {
					who = ROOT
					modifier = pol_declined_wedding
				}
			}
		}
		if = {
			limit = {
				has_consort = no
			}
			define_consort = {
				ADM = 2
				MIL = 2
				DIP = 2
			}
		}
	}
}

# Adopt Polish Culture event
country_event = {
	id = foi.18
	title = foi.eventname18
	desc = foi.eventdesc18
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes

	option = { #Adopt Polish Culture
		name = foi.event18opt1
		ai_chance = { factor = 100 }
		set_ruler_culture = polish
		change_primary_culture = polish
		if = {
			limit = {
				has_heir = yes
			}
			set_heir_culture = polish
		}
		if = {
			limit = {
				culture_group = west_slavic
				has_consort = yes
			}
			set_consort_culture = polish
		}
	}
	option = { #Maintain current culture
		name = foi.event18opt2
		ai_chance = { factor = 0 }
		add_prestige = 10
		hidden_effect = {
			every_owned_province = {
				limit = {
					culture = polish
				}
				add_unrest = 5
			}
		}
	}
}
