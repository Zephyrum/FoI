#Old Basileus dies
country_event = {
	id = foi_flavor_byz_civ.1
	title = foi_flavor_byz_civ.1.title
	desc = {
		trigger = {
			has_heir_flag = sick_boi_basileus_heir
		}
		desc = foi_flavor_byz_civ.1.desc.a
	}
	desc = {
		trigger = {
			AND = {
				has_heir = no
				NOT = { has_heir_flag = sick_boi_basileus_heir }
			}
		}
		desc = foi_flavor_byz_civ.1.desc.b
	}
	desc = {
		trigger = {
			AND = {
				has_heir = yes
				NOT = { has_heir_flag = sick_boi_basileus_heir }
			}
		}
		desc = foi_flavor_byz_civ.1.desc.c
	}
	picture = WOUNDED_SOLDIERS_eventPicture
	
	fire_only_once = yes
	
	major = yes
	
	trigger = {
		tag = BYZ
		OR = {
			AND = {
				has_ruler_flag = old_healthy_baslieus
				is_year = 1455
			}
			NOT = { has_ruler_flag = old_healthy_baslieus }	
		}
	}
	
	mean_time_to_happen = {
		months = 60
		
		modifier = {
			factor = 0.001
			NOT = { has_ruler_flag = old_healthy_baslieus }
		}
	}
	
	immediate = {
		if = {
			limit = {
				OR = {
					has_ruler_flag = old_healthy_baslieus
					NOT = { has_ruler = "Konstantinos XI Andronikos" }
				}
			}
			kill_ruler = yes
			if = {
				limit = { NOT = { has_heir_flag = sick_boi_basileus_heir } }
				kill_heir = yes
			}
		}
	}
	
	option = { #still has sick boi
		name = foi_flavor_byz_civ.1.a
		ai_chance = { factor = 100 }
		trigger = {
			OR = {
				has_heir_flag = sick_boi_basileus_heir
				has_ruler = "Konstantinos XI Andronikos"
			}
		}
		
		clr_heir_flag = sick_boi_basileus_heir
		set_country_flag = sick_boi_basileus
		set_country_flag = court_in_constaniople
		
		hidden_effect = {
			country_event = { 
				id = foi_flavor_byz_civ.2
				days = 210
				random = 5
			}
		}
	}
	
	option = { #civil war starts
		name = foi_flavor_byz_civ.1.b
		ai_chance = { factor = 100 }
		trigger = {
			NOT = {
				OR = {
					has_heir_flag = sick_boi_basileus_heir
					has_ruler = "Konstantinos XI Andronikos"
				}
			}
		}
		
		define_ruler = {
			name = "Manuel Demetrius"
			dynasty = "Komnenos"
			adm = 3
			dip = 3
			mil = 6
			age = 25
			claim = 100
			fixed = yes
			male = yes
		}
		set_country_flag = court_in_constaniople
		set_ruler_flag = manuels_fallback_luck
		clr_country_flag = basileus_not_fallen_ill
		hidden_effect = {
			country_event = { 
				id = foi_flavor_byz_civ.3
				days = 1
				random = 20
			}
		}
	}
}
