country_event = {

	title = "foi_flavor_tou.1.title"
	desc = "foi_flavor_tou.1.desc"
	picture = RELIGIOUS_CONVERSION_eventPicture
	
	fire_only_once = yes
	
	is_triggered_only = yes
	
	major = yes
	
	option = { # Stay Catholic
		name = "foi_flavor_tou.1.a"
		ai_chance = { factor = 50 }
		trigger = { religion = catholic }
		PAP = {
			add_opinion = {
				who = TOU
				modifier = opinion_loyal_to_pope
			}
		}
		add_prestige = 50
		add_papal_influence = 100
	}
	option = { # Become Protestant
		name = "foi_flavor_tou.1.b"
		ai_chance = { factor = 25 }
		change_religion = protestant
		PAP = {
			add_opinion = {
				who = PAP
				modifier = opinion_disappointed_pope
			}
		}
		if = {
			limit = {
				NOT = {
					calc_true_if = {
						all_province = {
							religion = protestant
							is_reformation_center = yes
						}
						amount = 3
					}
				}
			}
			hidden_effect = {
				random_owned_province = {
					limit = {
						can_have_center_of_reformation_trigger = {
							RELIGION = protestant
						}
					}
					change_religion = protestant
					add_reform_center = protestant
				}
			}
		}
		random_owned_province = {
			change_religion = protestant
		}
		random_owned_province = {
			NOT = { religion = protestant}
			spawn_rebels = {
				size = 1
				type = catholic_rebels
			}
		}
		add_prestige = 100
		add_church_power = 50
	}
	option = { # Become Cathar
		name = "foi_flavor_tou.1.c"
		change_religion = cathar
		PAP = {
			add_opinion = {
				who = PAP
				modifier = opinion_upset_pope
			}
		}
		capital = {
			change_religion = cathar
		}
		4695 = {
			change_religion = cathar
			add_reform_center = cathar
		} 
		random_owned_province = {
			NOT = { religion = cathar}
			change_religion = cathar
		}
		random_owned_province = {
			NOT = { religion = cathar}
			spawn_rebels = {
				size = 1
				type = catholic_rebels
			}
		}
		hidden_effect = {
			4695 = {
				province_event = {
					id = flavor_tou.13
					days = 36500
				}
			}
		}
		add_prestige = 100
		add_church_power = 50
	}
}
