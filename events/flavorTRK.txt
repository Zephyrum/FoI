# Demand Fealty
country_event = {
	id = foi_flavor_trk.1
	title = "foi_flavor_trk.1.title"
	desc = "foi_flavor_trk.1.desc"
	picture = CONQUEST_eventPicture
	
	fire_only_once = yes
	
	is_triggered_only = yes
	
	option = { # Demand Fealty
		name = "foi_flavor_trk.1.a"
		ai_chance = { factor = 100 }
		
		if = {
			limit = {
				exists = PLA
			}
			TRK = {
				add_opinion = {
					who = PLA
					modifier = opinion_united_dynasties
				}
			}
			PLA = {
				add_opinion = {
					who = TRK
					modifier = opinion_united_dynasties
				}
			}
			create_union = PLA
		}
		if = {
			limit = {
				exists = KLV
			}
			TRK = {
				add_opinion = {
					who = KLV
					modifier = opinion_united_dynasties
				}
			}
			KLV = {
				add_opinion = {
					who = TRK
					modifier = opinion_united_dynasties
				}
			}
			create_union = KLV
		}
	}
	option = { # Don't?????
		name = "foi_flavor_trk.1.b"
		ai_chance = { factor = 0 }
		
	}
}
