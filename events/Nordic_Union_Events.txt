#The Nordic Union
country_event = {
	id = foi_nordic_union.1
	title = "foi_nordic_union.1.t"
	desc = "foi_nordic_union.1.d"
	picture = DIPLOMACY_eventPicture

	major = yes
	major_trigger = {
		provinces_on_capital_continent_of = SWE
	}
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			AND = { #Sweden ruler/Norway is AI
				tag = SWE
				has_ruler = "Erik XV Lukasson"
				is_subject = no
				NOR = {
					ai = yes
					has_heir = "Erik Lukasson"
					is_subject = no
				}
			}
			AND = { #Norway ruler/Sweden is AI
				tag = NOR
				has_ruler = "Erik III Lukasson"
				is_subject = no
				SWE = {
					ai = yes
					has_heir = "Erik Lukasson"
					is_subject = no
				}
			}
			AND = { #Norway ruler/Sweden is player
				tag = SWE
				ai = no
				has_heir = "Erik Lukasson"
				is_subject = no
				NOR = {
					has_ruler = "Erik XV Lukasson"
					is_subject = no
				}
			}
			AND = { #Sweden ruler/Norway is player
				tag = NOR
				ai = no
				has_heir = "Erik Lukasson"
				is_subject = no
				SWE = {
					has_ruler = "Erik III Lukasson"
					is_subject = no
				}
			}
			AND = {
				is_year = 1456
				has_country_flag = force_nordic_union_flag
				has_heir = "Erik Lukasson"
				is_subject = no
				any_country = {
					has_heir = "Erik Lukasson"
					is_subject = no
				}
			}
		}
	}
	
	immediate = {
		SWE = {
			clr_country_flag = erik_heir_of_sweden_flag
			clr_country_flag = erik_king_of_sweden_flag
		}
		NOR = {
			clr_country_flag = erik_heir_of_norway_flag
			clr_country_flag = erik_king_of_norway_flag
		}
	}

	mean_time_to_happen = {
		months = 120
	}
	
	option = {	# Player-option
		name = "foi_nordic_union.1.a"
		ai_chance = { factor = 100 }
		trigger = {
			OR = {
				AND = {
					tag = SWE
					ai = no
					NOR = { ai = yes }
				}
				AND = {
					tag = NOR
					ai = no
					SWE = { ai = yes }
				}
			}
		}
		if = {
			limit = { tag = NOR }
			if = {
				limit = { has_ruler  = "Kristina I" }
				kill_ruler = yes
			}
			create_union = SWE
			add_opinion = {
				who = SWE
				modifier = nordic_union
			}
			SWE = {
				add_opinion = {
					who = ROOT
					modifier = nordic_union
				}
				add_liberty_desire = -50
				random_owned_province = {
					limit = {
						is_capital = no
					}
					spawn_rebels = {
						type = pretender_rebels
						size = 3
					}
				}
			}
			every_known_country = {
				limit = {
					alliance_with = SWE
				}
				add_opinion = {
					who = ROOT
					modifier = nordic_wedding_negative
				}
			}
		}
		if = {
			limit = { tag = SWE }
			if = {
				limit = { has_ruler  = "Karl VIII" }
				kill_ruler = yes
			}
			create_union = NOR
			add_opinion = {
				who = NOR
				modifier = nordic_union
			}
			NOR = {
				add_opinion = {
					who = ROOT
					modifier = nordic_union
				}
				add_liberty_desire = -50
				if = {
					limit = { senior_union_with = ICE }
					hidden_effect = { ICE = { restore_country_name = yes } }
					inherit = ICE
				}
				random_owned_province = {
					limit = {
						is_capital = no
					}
					spawn_rebels = {
						type = pretender_rebels
						size = 3
					}
				}
			}
			every_known_country = {
				limit = {
					alliance_with = NOR
				}
				add_opinion = {
					who = ROOT
					modifier = nordic_wedding_negative
				}
			}
		}
	}
	
	option = {	# AI-option
		name = "foi_nordic_union.1.b"
		ai_chance = { factor = 100 }
		trigger = {
			AND = {
				SWE = { ai = yes }
				NOR = { ai = yes }
			}
		}
		if = {
			limit = { tag = NOR }
			if = {
				limit = { has_ruler  = "Kristina I" }
				kill_ruler = yes
			}
			create_union = SWE
			add_opinion = {
				who = SWE
				modifier = nordic_union
			}
			SWE = {
				add_opinion = {
					who = ROOT
					modifier = nordic_union
				}
				add_liberty_desire = -50
				random_owned_province = {
					limit = {
						is_capital = no
					}
					spawn_rebels = {
						type = pretender_rebels
						size = 3
					}
				}
			}
			every_known_country = {
				limit = {
					alliance_with = SWE
				}
				add_opinion = {
					who = ROOT
					modifier = nordic_wedding_negative
				}
			}
		}
		if = {
			limit = { tag = SWE }
			if = {
				limit = { has_ruler  = "Karl VIII" }
				kill_ruler = yes
			}
			create_union = NOR
			add_opinion = {
				who = NOR
				modifier = nordic_union
			}
			NOR = {
				add_opinion = {
					who = ROOT
					modifier = nordic_union
				}
				add_liberty_desire = -50
				if = {
					limit = { senior_union_with = ICE }
					hidden_effect = { ICE = { restore_country_name = yes } }
					inherit = ICE
				}
				random_owned_province = {
					limit = {
						is_capital = no
					}
					spawn_rebels = {
						type = pretender_rebels
						size = 3
					}
				}
			}
			every_known_country = {
				limit = {
					alliance_with = NOR
				}
				add_opinion = {
					who = ROOT
					modifier = nordic_wedding_negative
				}
			}
		}
	}
}

#Union Forcer
country_event = {
	id = foi_nordic_union.2
	title = none
	desc = none
	picture = none
	hidden = yes	

	fire_only_once = yes
	
	trigger = {
		tag = SWE
		has_heir = "Erik Lukasson"
		NOR = { has_heir = "Erik Lukasson" }
		OR = {
			ai = yes
			NOR = { ai = yes }
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		if = {	#SWE - Player-option
			limit = {
				ai = no
				NOR = { ai = yes }
			}
			hidden_effect = {
				set_country_flag = force_nordic_union_flag
				set_heir_culture = swedish
				NOR = { set_heir_culture = swedish }
			}
		}
		else_if = {	#NOR - Player-option
			limit = {
				ai = yes
				NOR = { ai = no }
			}
			hidden_effect = {
				NOR = { set_country_flag = force_nordic_union_flag }
				set_heir_culture = norwegian
				NOR = { set_heir_culture = norwegian }
			}
		}
		else_if = {	#AI-option
			limit = {
				ai = yes
				NOR = { ai = yes }
			}
			hidden_effect = {
				random_list = {
					65 = {
						set_country_flag = force_nordic_union_flag
						set_heir_culture = swedish
						NOR = { set_heir_culture = swedish }
					}
					35 = {
						NOR = { set_country_flag = force_nordic_union_flag }
						set_heir_culture = norwegian
						NOR = { set_heir_culture = norwegian }
					}
				}
			}
		}
	}
	
	option = { }
}

#Double Player Solver - Sweden
country_event = {
	id = foi_nordic_union.100
	title = "foi_nordic_union.100.t"
	desc = "foi_nordic_union.100.d"
	picture = DIPLOMACY_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = SWE
		has_heir = "Erik Lukasson"
		ai = no
		NOR = {
			ai = no
			has_heir = "Erik Lukasson"
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		SWE = {
			clr_country_flag = erik_heir_of_sweden_flag
			clr_country_flag = erik_king_of_sweden_flag
		}
		NOR = {
			clr_country_flag = erik_heir_of_norway_flag
			clr_country_flag = erik_king_of_norway_flag
		}
	}
	
	option = {	# Player-option
		name = "foi_nordic_union.100.a"
		ai_chance = { factor = 50 }
		
		kill_heir = { allow_new_heir = no }
		define_heir = {
			dynasty = ROOT
			age = 0
			adm = 6
			dip = 6
			mil = 6
			name = "Frej"
			claim = 100
			no_consort_with_heir = yes
		}
		
		hidden_effect = { country_event = { id = foi_nordic_union.101 } }
		
		NOR = {
			kill_heir = { allow_new_heir = no }
			define_heir = {
				dynasty = ROOT
				age = 0
				adm = 6
				dip = 6
				mil = 6
				name = "Froya"
				claim = 100
				female = yes
				no_consort_with_heir = yes
			}
		}
	}
	
}
