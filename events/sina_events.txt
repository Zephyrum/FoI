country_event = {
	
	id = sina.1
	title = "sina.1t"
	desc = "sina.1d"
	picture = ACCUSATION_eventPicture
	
	trigger = {
		tag = YUA
		NOT = { heir_culture = chihan }
	}
	
	option = {
		name = "sina.1a" # Take control of education
		
			add_stability = -1

			add_dip_power = -50
			add_mil_power = -50
			add_legitimacy = -10
			change_heir_adm = 1
			change_heir_dip = 1
			change_heir_mil = 1
	}
	
	option = {
		name = "sina.1b" # Leave it to the court
		
		set_heir_culture = chihan
	}
	
	mean_time_to_happen = {
		months = 36
	}
}

country_event = { #Warlords!
	
	id = sina.3
	title = "sina.3t"
	desc = "sina.3d"
	picture = ACCUSATION_eventPicture
	fire_only_once = yes
	major = yes


	trigger = {
		tag = YUA
		1816 = {
			NOT = { controlled_by = YUA }
		}
	}
	
	option = {
		name = "sina.3a" #Challenge accepted, because you don't really have an alternative
		
			every_province = {
				limit = {
					superregion = china_superregion
				}
				add_permanent_province_modifier = { 
					name = china_is_contested
					duration = 18250
				}
			}
			every_country = { 
    			limit = {
					OR = {
        				has_reform = warlord_state
						has_reform = warlord_republic
					}
    			}
    			add_country_modifier = { 
        			name = warlord_era_of_china
        			duration = 18250
    			}
			}
			MNG = {
				country_event = {
    				id = sina.4
				}
			}
	}
	
	mean_time_to_happen = {
		days = 1
	}
}
