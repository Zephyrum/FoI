namespace = saared

# Schism Mended (for country that passed it)
country_event = {
	
	id = saared.1
	title = "saared.eventname1"
	desc = "saared.eventdesc1"
	picture = ORTHODOX_eventPicture

	fire_only_once = yes

	trigger = {
		if = {
			tag = BYZ
			is_subject = no
			religion = orthodox
			NOT = { dynasty = "Palaiologos" }
			NOT = { legitimacy = 30	}
			any_owned_province = {
				development = 10
				is_capital = no
			}
			num_of_cities = 10
			OR = {
				151 = {
					NOT = { controlled_by = BYZ }
				}	
				151 = {
					NOT = { owned_by = BYZ }
				}
			}			
			NOT = { is_year = 1500 }
		}
	}

	mean_time_to_happen = {
		months = 1
	}
	
	option = {
		name = "saared.event1opt1" # Leo XI becomes emperor
		ai_chance = { factor = 0.8 }
		define_ruler = {
			name = "Leo XI"
			dynasty = "Palaiologos"
			culture = greek
			religion = iconoclast
			fixed = yes
			adm = 6
			dip = 6
			mil = 6
			claim = 40
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality = zealot_personality }
		}
		change_religion = iconoclast
		add_country_modifier = {
			name = "conversion_zeal"
			duration = 3650
		}
		if = {
			limit = {
				has_heir = yes
			}
			kill_heir = yes
		}
		add_stability = -1
	}
	option = { #Leo XI has to fight for it then
		name = "saared.event1opt2"
		ai_chance = { factor = 0.2 }
		add_prestige = 30
		set_country_flag = saared_iconoclast_rebellion
		random_owned_province = {
			limit = {
				is_capital = no
				development = 10
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 3
				leader = "Leo XI Palaiologos"
				culture = greek
				religion = iconoclast
			}
		}
	}
}
