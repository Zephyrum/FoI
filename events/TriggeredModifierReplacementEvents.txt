#Discovery of the East Indian trade route
country_event = {
	id = triggered_modifier_replacement.1
	title = "triggered_modifier_replacement.title"
	desc = "triggered_modifier_replacement.1.d"
	picture = PORTUGUESE_MERCHANTS_eventPicture

	trigger = {
		ai = no
		NOT = { has_country_modifier = east_indian_trade_route }
		
		NOT = {
			capital_scope = {
				continent = asia
			}
		}
		NOT = {
			capital_scope = {
				continent = oceania
			}
		}
		
		has_discovered = 1460 # Cape of Good Hope
		num_of_ports = 4
		has_idea = quest_for_the_new_world
		any_active_trade_node = {
			has_merchant = ROOT
			OR = {
				province_id = 1361 # Straits of Johor
				province_id = 1343 # Ganges Delta
				province_id = 504 # Sind
				province_id = 1340 # Comorin Cape
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = "east_indian_trade_route"
				duration = -1
			}
		}
	}
	
	option = {
		name = "triggered_modifier_replacement_getting_good"
		tooltip = {
			add_country_modifier = {
				name = "east_indian_trade_route"
				duration = -1
			}
		}
	}
}
