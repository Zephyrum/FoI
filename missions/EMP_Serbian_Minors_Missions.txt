emp_ser_minor_economy_internal = {
	slot = 1
	generic = no
	ai = yes
	potential_on_load = {
		has_dlc = "Emperor"
	}
	potential = {
		has_dlc = "Emperor"
		OR = {
			tag = BEL
			tag = MON
		}
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	emp_ser_law_of_the_pious = {
		icon = law_of_the_pious
		required_missions = { }
		position = 1
		
		trigger = {
			if = {
				limit = { 
					has_estate = estate_burghers 
				}
				estate_loyalty = {
					estate = estate_burghers
					loyalty = 60
				}
			}
			if = {
				limit = {
					has_estate = estate_church
				}
				estate_loyalty = {
					estate = estate_church
					loyalty = 60
				}
			}
			if = {
				limit = {
					has_estate = estate_nobles
				}
				estate_loyalty = {
					estate = estate_nobles
					loyalty = 60
				}
			}
			if = {
				limit = {
					NOT = { has_estate = estate_burghers }
					NOT = { has_estate = estate_church }
					NOT = { has_estate = estate_nobles }
				}
				NOT = { average_unrest = 1 }
			}
		}
		effect = {
			add_stability = 1
			if = { 
				limit = { NOT = { legitimacy = 80 } }
				add_legitimacy = 20
			}
			add_prestige = 20
		}
	}
	
	emp_ser_reissue_the_novo_brdo_code = {
		icon = reissue_the_mining_code
		required_missions = { emp_ser_law_of_the_pious }
		position = 2
	
		trigger = {
			grown_by_development = 20
			any_owned_province = {
				OR = {
					has_building = workshop
					has_building = counting_house
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "ser_novo_brdo_code"
				duration = 7300
			}
		}
	}
}

		required_missions = { }
		position = 1
		
		provinces_to_highlight = {
		}
		trigger = {
			if = {
				limit = { 
					religion = orthodox 
					has_dlc = "Third Rome"
				}
				custom_trigger_tooltip = {
					tooltip = ser_has_any_icon
					OR = {
						current_icon = icon_michael
						current_icon = icon_eleusa
						current_icon = icon_pancreator
						current_icon = icon_nicholas
						current_icon = icon_climacus
					}
				}
			}
			else = {
				calc_true_if = {
					all_owned_province = {
						OR = {
							has_building = temple
							has_building = cathedral
						}
					}
					amount = 3
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "ser_caesaropapism"
				duration = 7300
			}
		}
	}
	
	emp_ser_promote_the_patriarchate = {
		icon = promote_the_patriarchate
		required_missions = { emp_ser_reform_the_nomocanon }
		position = 3
		
		provinces_to_highlight = {
			ROOT = { NOT = { religion = orthodox } }
			owner = { tag = ROOT }
			NOT = { religion = ROOT }
		}
		trigger = {

				limit = { religion = orthodox }

			else = {

		}
		effect = {

					duration = 7300
				}
			}
			else = {
				add_adm_power = 100
			}
		}
	}
	
	emp_ser_guardians_of_st_sava = {
