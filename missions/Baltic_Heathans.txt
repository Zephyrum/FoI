pth_slot1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = PTH
		}
	}
	has_country_shield = yes
	
	pth_kill_the_opressors = {
		position = 1
		icon = mission_teutonic_knights
		required_missions = { }
		provinces_to_highlight = {
			OR = {
				area = east_prussia_area
				province_id = 1841
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		
		trigger = {
			east_prussia_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
			owns_core_province = 1841
			NOT = { exists = TEU }
		}
		
		effect = {
			add_country_modifier = {
				name = "heathens_crushed_modifier"
				duration = 18250  #50 Jahre
			}
		}
	}
	
	
	pth_liberate_the_north = {
		position = 2
		icon = mission_conquer_5_states
		required_missions = { pth_kill_the_opressors }
		provinces_to_highlight = { }
		
		trigger = {
			religious_unity = 1
			army_size_percentage = 0.8
			manpower_percentage = 0.80
		}
		
		effect = {
			every_province = {
				limit = {
					owned_by = LIV
				}
				add_permanent_claim = ROOT
			}
			country_event = {
				id = foi_flavor_baltic.3
			}
		}
	}
	
	pth_last_crusader_stronghold = {
		position = 3
		icon = mission_hanseatic_city
		required_missions = { pth_liberate_the_north }
		provinces_to_highlight = { 
		OR = {
				owned_by = LIV
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}

		trigger = {
			NOT = { exists = LIV }
			OR = {
				HSA = {
					has_opinion = {
						who = ROOT
						value = 100
					}
				}
				NOT = { exists = HSA}
			}
		}
		
		effect = {
			add_permanent_claim = 38
			if = {
				limit = {
					exists = HSA
				}
				add_treasury = -250
				HSA = {
					country_event = {
						id = foi_flavor_baltic.4
					}
				}
			}   
		}
	}
	
	pth_dominate_the_baltic = {
		position = 4
		icon = mission_eastern_european_city
		required_missions = { pth_last_crusader_stronghold }
		provinces_to_highlight = {
			OR = {
				area = east_prussia_area
				area = curonia_area
				area = livonia_area
				area = estonia_ingria_area
				province_id = 1841
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		
		trigger = {
			east_prussia_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
			curonia_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
			livonia_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
			estonia_ingria_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
				
			}
			1841 = {
				country_or_non_sovereign_subject_holds = ROOT
			}
			religious_unity = 1
		}
		
		effect = {
			add_country_modifier = {
				name = "for_the_old_gods_modifier"
				duration = -1 #end of game
			}
			baltic_region = {
				add_permanent_claim = ROOT
			}
		}
	}
