			any_ally = {
				culture_group = south_slavic

				is_great_power = yes
			}
			OR = {
				alliance_with = BOH
				alliance_with = BYZ
				alliance_with = BUL

			}
		}
		effect = {
			add_country_modifier = {
				name = "albanian_diplomacy"
				duration = 7300

			every_ally = {
				add_opinion = {
					who = ROOT
					modifier = opinion_pap_albania
				}
				add_trust = {
					who = THIS
					value = 25
					mutual = yes
				}
			}
		}
	}
	
	alb_inspire_resistance = {
		icon = mission_alliances
		required_missions = { alb_withstand_the_turks }
		provinces_to_highlight = {
			AND = {
				owner = { alliance_with = ALB }
				is_capital = yes
			}
		}
		trigger = {
			calc_true_if = {
				all_country = {
					AND = {
						alliance_with = ROOT
						has_opinion = {
							who = ROOT
							value = 200
						}
					}
				}
				amount = 6
			}
		}
		effect = {
			add_country_modifier = {
				name = "albanian_diplomacy"
				duration = -1
			}
			add_country_modifier = {
				name = "alb_inspire_resistance"
				duration = -1
			}
		}
	}
	
	alb_lord_of_albania = {
		icon = capture_adriatic_ports
		required_missions = { }
